**1. 任务管理存在竞态条件**

- **位置**: [src/store/taskStore.ts](../../src/store/taskStore.ts#L137-L144)

- **问题描述**: 
  ```typescript
  async function tryStartNext() {
    if (runningCount.value >= maxConcurrent.value) return;
    const nextTask = tasks.value.find((t) => t.status === TaskStatus.Pending);
    if (nextTask) {
      await startTask(nextTask.id);
    }
  }
  ```

- **影响**: 
  - 在并发场景下可能启动超过 `maxConcurrent` 个任务
  - 没有锁机制保护临界区

- **建议**: 
  ```typescript
  // 使用锁或标志位保护
  let isStarting = false;
  
  async function tryStartNext() {
    if (isStarting) return;
    isStarting = true;
    
    try {
      if (runningCount.value >= maxConcurrent.value) return;
      const nextTask = tasks.value.find((t) => t.status === TaskStatus.Pending);
      if (nextTask) {
        await startTask(nextTask.id);
      }
    } finally {
      isStarting = false;
    }
  }
  ```

- **位置**: [src-tauri/src/modules/task_manager.rs](../../src-tauri/src/modules/task_manager.rs#L49-L73)

- **问题描述**: 
  ```rust
  std::thread::spawn(move || {
      if let Some(stderr) = child.stderr.take() {
          let reader = BufReader::new(stderr);
          for line in reader.lines() {
              if let Ok(line) = line {
                  if let Ok(progress) = parse_progress(&line) {
                      let _ = app_clone.emit("ffmpeg-progress", serde_json::json!({
  ```

  ```rust

      match child.wait_timeout(timeout) {
          Ok(Some(status)) => {
              // 正常结束
          }
          Ok(None) => {
              // 超时，强制杀死进程
              let _ = child.kill();
          }
          Err(e) => {
              // 错误处理
          }
      }
  });
  ```

- **位置**: [src-tauri/src/lib.rs](../../src-tauri/src/lib.rs#L160-L191)

- **问题描述**: 
  ```rust
  #[tauri::command]
  fn extract_video_frame(video_path: String, timestamp: String) -> Result<String, String> {
      // ...
      let status = Command::new("ffmpeg")
          .args(&[
  ```

  ```rust
      let path = PathBuf::from(path);
      
      // 检查文件是否存在
      if !path.exists() {
          return Err("文件不存在".to_string());
      }
      
      // 检查文件扩展名
      let ext = path.extension()
          .and_then(|s| s.to_str())
          .unwrap_or("");
      
      const ALLOWED_EXTENSIONS: &[&str] = &["mp4", "mkv", "avi", "mov", "webm"];
      if !ALLOWED_EXTENSIONS.contains(&ext) {
          return Err("不支持的文件格式".to_string());
      }
      
      Ok(path)
  }
  ```

  ```rust
      match Command::new(&ffmpeg_path).arg("-version").output() {
          Ok(output) => {
              if output.status.success() {
                  let version_info = String::from_utf8_lossy(&output.stdout);
                  let first_line = version_info.lines().next().unwrap_or("FFmpeg 可用");
                  Ok(first_line.to_string())
              } else {
                  Err("FFmpeg 执行失败".to_string())
              }
          }
          Err(e) => Err(format!("未找到 FFmpeg: {}", e)),
      }
  }
  ```

- **影响**: 
  - 错误信息可能包含系统路径等敏感信息

- **建议**: 
  ```rust
  Err(e) => Err("未找到 FFmpeg，请检查路径是否正确".to_string()),
  ```
